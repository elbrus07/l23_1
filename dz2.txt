#include <iostream>
using namespace std;

int main() {
    
    // ЗАДАЧА 1
    cout << "=== Задача 1 ===" << endl;
    int n;
    cout << "Введите число: ";
    cin >> n;
    
    // for
    int temp = n;
    int max_d = 0, min_d = 9;
    for (; temp > 0; temp = temp / 10) {
        int d = temp % 10;
        if (d > max_d) max_d = d;
        if (d < min_d) min_d = d;
    }
    cout << "FOR: max=" << max_d << " min=" << min_d << " proizv=" << max_d * min_d << endl;
    
    // while
    temp = n;
    max_d = 0; min_d = 9;
    while (temp > 0) {
        int d = temp % 10;
        if (d > max_d) max_d = d;
        if (d < min_d) min_d = d;
        temp = temp / 10;
    }
    cout << "WHILE: max=" << max_d << " min=" << min_d << " proizv=" << max_d * min_d << endl;
    
    // do while
    temp = n;
    max_d = 0; min_d = 9;
    do {
        int d = temp % 10;
        if (d > max_d) max_d = d;
        if (d < min_d) min_d = d;
        temp = temp / 10;
    } while (temp > 0);
    cout << "DO WHILE: max=" << max_d << " min=" << min_d << " proizv=" << max_d * min_d << endl;
    
    // ЗАДАЧА 2
    cout << "\n=== Задача 2 ===" << endl;
    int x, s = 0;
    cout << "Введите числа (0 конец): ";
    
    // for
    for (cin >> x; x != 0; cin >> x) {
        s = s + x;
    }
    cout << "FOR: summa=" << s << endl;
    
    // while
    s = 0;
    cin >> x;
    while (x != 0) {
        s = s + x;
        cin >> x;
    }
    cout << "WHILE: summa=" << s << endl;
    
    // do while
    s = 0;
    do {
        cin >> x;
        s = s + x;
    } while (x != 0);
    cout << "DO WHILE: summa=" << s << endl;
    
    // ЗАДАЧА 3
    cout << "\n=== Задача 3 ===" << endl;
    int chislo, cifra;
    cout << "Введите число: ";
    cin >> chislo;
    cout << "Введите цифру для удаления: ";
    cin >> cifra;
    
    // for
    temp = chislo;
    int res = 0, mn = 1;
    for (; temp > 0; temp = temp / 10) {
        int d = temp % 10;
        if (d != cifra) {
            res = res + d * mn;
            mn = mn * 10;
        }
    }
    cout << "FOR: result=" << res << endl;
    
    // while
    temp = chislo;
    res = 0; mn = 1;
    while (temp > 0) {
        int d = temp % 10;
        if (d != cifra) {
            res = res + d * mn;
            mn = mn * 10;
        }
        temp = temp / 10;
    }
    cout << "WHILE: result=" << res << endl;
    
    // do while
    temp = chislo;
    res = 0; mn = 1;
    do {
        int d = temp % 10;
        if (d != cifra) {
            res = res + d * mn;
            mn = mn * 10;
        }
        temp = temp / 10;
    } while (temp > 0);
    cout << "DO WHILE: result=" << res << endl;
    
    // ЗАДАЧА 4
    cout << "\n=== Задача 4 ===" << endl;
    int N, M;
    cout << "Введите N: ";
    cin >> N;
    cout << "Введите M: ";
    cin >> M;
    
    // for
    cout << "FOR:" << endl;
    for (int i = N; i <= M; i++) {
        cout << i << ": ";
        for (int j = 2; j < i; j++) {
            if (i % j == 0) {
                cout << j << " ";
            }
        }
        cout << endl;
    }
    
    // while
    cout << "WHILE:" << endl;
    int i = N;
    while (i <= M) {
        cout << i << ": ";
        int j = 2;
        while (j < i) {
            if (i % j == 0) {
                cout << j << " ";
            }
            j++;
        }
        cout << endl;
        i++;
    }
    
    // do while
    cout << "DO WHILE:" << endl;
    i = N;
    do {
        cout << i << ": ";
        int j = 2;
        do {
            if (i % j == 0) {
                cout << j << " ";
            }
            j++;
        } while (j < i);
        cout << endl;
        i++;
    } while (i <= M);
    
    // ЗАДАЧА 5
    cout << "\n=== Задача 5 ===" << endl;
    
    // for
    cout << "FOR:" << endl;
    for (int i = 1; i <= 10; i++) {
        for (int j = 1; j <= 10; j++) {
            cout << i * j << " ";
        }
        cout << endl;
    }
    
    // while
    cout << "WHILE:" << endl;
    i = 1;
    while (i <= 10) {
        int j = 1;
        while (j <= 10) {
            cout << i * j << " ";
            j++;
        }
        cout << endl;
        i++;
    }
    
    // do while
    cout << "DO WHILE:" << endl;
    i = 1;
    do {
        int j = 1;
        do {
            cout << i * j << " ";
            j++;
        } while (j <= 10);
        cout << endl;
        i++;
    } while (i <= 10);
    
    return 0;
}
